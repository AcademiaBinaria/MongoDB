## 06.7 aggregate()



```javascript

// Lo primero ser√° convertir las cadenas de texto a fechas reales

db.tickets.find().forEach(function(ticket){
  ticket.hora = ISODate(ticket.hora);
  db.tickets.save(ticket);
})


// $date_operators
db.tickets.aggregate({
    $project:{
        hora:1,
        year:{$year:"$hora"},
        month:{$month:"$hora"},
        dayOfMonth: {$dayOfMonth:"$hora"},
        dayOfYear: {$dayOfYear:"$hora"},
        dayOfWeek: {$dayOfWeek:"$hora"},
        week:{$week:"$hora"},
        hour:{$hour:"$hora"}
    }
});

```
